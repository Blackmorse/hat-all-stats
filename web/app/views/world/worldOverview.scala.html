@import service.OverviewStatistics @import service.LeaguesInfo @import com.blackmorse.hattrick.api.worlddetails.model.League @(overviewStatistics: OverviewStatistics, league: League, leaguesInfo: LeaguesInfo, pageSize: Int)(implicit messages: Messages) @main(messages("overview.world_overview"), world.header(leaguesInfo), world.leftBar(leaguesInfo), messages("overview.world_overview")) {
@world.summaryStats(overviewStatistics.summaryOverview)
@world.formationsTable(overviewStatistics.formations)
@world.averageOverview(overviewStatistics.averageOverview, league)
@world.matchStatsTable(overviewStatistics.topRandomMatches, messages("table.hatstats"), leaguesInfo, messages("overview.surprising_matches"))
@world.teamStatsTable(overviewStatistics.topHatstatsTeams, messages("table.hatstats"), leaguesInfo, messages("overview.top_teams"), i => Html(i.toString), (leagueId, parameters) => routes.LeagueController.bestTeams(leagueId, parameters), "hatstats", pageSize)
@world.teamStatsTable(overviewStatistics.topSalaryTeams, messages("table.salary") + "," + (if(league.getCountry.getCurrencyName == null) "$" else league.getCountry.getCountryName), leaguesInfo, messages("overview.top_salary_teams"), salary => common.formatters.salaryFormatter(salary, league), (leagueId, parameters) => routes.LeagueController.teamState(leagueId, parameters), "salary", pageSize)
@world.matchStatsTable(overviewStatistics.topHatstatsMatches, messages("table.hatstats"), leaguesInfo, messages("overview.top_matches"))
@world.playersStatsTable(overviewStatistics.topSalaryPlayers, messages("table.salary") + "," + (if(league.getCountry.getCurrencyName == null) "$" else league.getCountry.getCountryName), leaguesInfo, messages("overview.top_salary_players"), salary => common.formatters.salaryFormatter(salary, league), (leagueId, statisticsParameters) => routes.LeagueController.playerState(leagueId, statisticsParameters), "salary", pageSize)
@world.playersStatsTable(overviewStatistics.topRatingPlayers, messages("table.rating"), leaguesInfo, messages("overview.top_rating_players"), stars => common.formatters.ratingFormatter(stars), (leagueId, statisticsParameters) => routes.LeagueController.playerState(leagueId, statisticsParameters), "rating", pageSize)
}
